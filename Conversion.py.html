from tkinter import *
def button():
    mile_entry = entry_int.get()
    km_output = mile_entry * 1.60934
    output_string.set(km_output)

window = Tk()
window.geometry('400x200')
window.title('Conversion')
window.config(bg='black')

title_label = Label(window,
                    text='Miles to Kilometer',
                    font='Arial 20 bold',
                    fg='white',
                    bg='black')
title_label.pack()

frame = Frame(window)

entry_int = IntVar()
entry = Entry(frame,
              textvariable=entry_int)
button = Button(frame,
                text='Convert',
                bg='blue',
                fg='white',
                command=button)
entry.pack(side='left',padx=10)
button.pack(side='left')
frame.pack(pady=10)

output_string = StringVar()
output = Label(window,
               text='Output',
               font='Arial 20',
               textvariable=output_string,
               fg='white',
               bg='black')
output.pack()

window.mainloop()
